一、核心报错解读

错误原因
编码/乱码问题：禁忌、不è 是中文注释出现了编码错乱，MySQL 解析失败
换行/空格问题：SQL 语句换行时可能混入了不可见字符（如全角空格、换行符）
字符集不兼容：部分 MySQL 版本对中文注释的支持需要严格的 utf8mb4 配置

二、MySQL vs SQLite 最简单区别
1）MySQL（你们项目必须用这个）
是独立数据库服务，需要安装、启动
支持多人连接、高并发、正式项目、后端对接
适合：SpringBoot 后端 + 鸿蒙前端 + 爬虫入库
老师/实训项目标准选型
2）SQLite（轻量单机文件库）
只是一个文件，不用安装、不用启动
只能单机用，不支持多人同时写
适合：APP 本地缓存、小工具、测试
一句话总结
后端 + 爬虫 + 鸿蒙展示 → 用 MySQL
APP本地存点简单记录 → 才用 SQLite

def parse_hospital_type(type_str):
    """从高德type字段拆分医院类型和等级"""
    if not type_str:
        return "", ""
    parts = type_str.split(";")
    # 提取等级（如三级甲等）
    level_match = re.search(r"(三级甲等|三级乙等|二级甲等|二级乙等)", type_str)
    level = level_match.group() if level_match else ""
    # 提取类型（如综合医院）
    dept_type = parts[-1] if parts else ""
    return dept_type, level
def crawl_hospital_base(city="北京市"):

package com.example.medical.common;
import io.jsonwebtoken.Claims;import io.jsonwebtoken.Jwts;import io.jsonwebtoken.security.Keys;import org.springframework.stereotype.Component;
import javax.crypto.SecretKey;import java.util.Date;
@Componentpublic class JwtUtil {
    private static final String SECRET = "your_medical_project_secret_123456";
    private static final long EXPIRE = 1000 * 60 * 60 * 24;
    private final SecretKey key = Keys.hmacShaKeyFor(SECRET.getBytes());
    public String generateToken(String username) {
        return Jwts.builder()
                .setSubject(username)
                .setExpiration(new Date(System.currentTimeMillis() + EXPIRE))
                .signWith(key)
                .compact();
    }
    public String getUsername(String token) {
        Claims claims = Jwts.parserBuilder()
                .setSigningKey(key)
                .build()
                .parseClaimsJws(token)
                .getBody();
        return claims.getSubject();
    }}

@Servicepublic class UserServiceImpl extends ServiceImpl<UserMapper, User> implements UserService {
    @Autowired
    private JwtUtil jwtUtil;
    @Override
    public Result login(LoginVO vo) {
        User user = lambdaQuery()
                .eq(User::getUsername, vo.getUsername())
                .one();
        if (user == null) {
            return Result.fail("用户不存在");
        }
        if (!user.getPassword().equals(vo.getPassword())) {
            return Result.fail("密码错误");
        }
        String token = jwtUtil.generateToken(user.getUsername());
        return Result.success(token);
    }}

package com.example.medical.config;
