// entry/src/main/ets/common/utils/RouterUtil.etsimport { router, RouterOptions } from '@kit.ArkUI';
/**
 * 路由工具类（API 22 专用）
 */export class RouterUtil {
  /**
   * 跳转到指定页面
   * @param url 页面路径，如 'pages/HomePage'
   * @param params 传递的参数（可选）
   */
  static push(url: string, params?: Record<string, any>): void {
    const options: RouterOptions = {
      url: url,
      params: params
    };
    router.pushUrl(options)
      .then(() => console.info(`路由跳转成功: ${url}`))
      .catch((err) => console.error(`路由跳转失败: ${url}, 错误: ${JSON.stringify(err)}`));
  }
  /**
   * 替换当前页面（无法返回）
   * @param url 页面路径
   * @param params 传递的参数（可选）
   */
  static replace(url: string, params?: Record<string, any>): void {
    const options: RouterOptions = {
      url: url,
      params: params
    };
    router.replaceUrl(options)
      .then(() => console.info(`路由替换成功: ${url}`))
      .catch((err) => console.error(`路由替换失败: ${url}, 错误: ${JSON.stringify(err)}`));
  }
  /**
   * 返回上一页
   */
  static back(): void {
    router.back()
      .then(() => console.info('返回上一页成功'))
      .catch((err) => console.error(`返回失败: ${JSON.stringify(err)}`));
  }
  /**
   * 获取页面传递的参数
   * @returns 参数对象
   */
  static getParams(): Record<string, any> {
    return router.getParams() || {};
  }}

// entry/src/main/ets/common/utils/HttpUtil.etsimport { http, HttpRequestOptions, HttpResponse } from '@kit.NetworkKit';import { AppStorage } from '@kit.ArkUI';import { BASE_URL, TIMEOUT } from '../constants/ApiConstants';
/**
 * 网络请求工具类（API 22 专用）
 */export class HttpUtil {
  // 全局配置
  private static readonly BASE_URL = BASE_URL;
  private static readonly TIMEOUT = TIMEOUT;
  /**
   * GET 请求
   * @param url 接口路径（不含BASE_URL）
   * @param params 查询参数
   * @returns Promise<数据>
   */
  static async get<T>(url: string, params?: Record<string, any>): Promise<T> {
    return this.request<T>(http.RequestMethod.GET, url, params);
  }
  /**
   * POST 请求
   * @param url 接口路径
   * @param data 请求体
   * @returns Promise<数据>
   */
  static async post<T>(url: string, data?: any): Promise<T> {
    return this.request<T>(http.RequestMethod.POST, url, data);
  }
  /**
   * PUT 请求
   */
  static async put<T>(url: string, data?: any): Promise<T> {
    return this.request<T>(http.RequestMethod.PUT, url, data);
  }
  /**
   * DELETE 请求
   */
  static async delete<T>(url: string, params?: Record<string, any>): Promise<T> {
    return this.request<T>(http.RequestMethod.DELETE, url, params);
  }
  /**
   * 统一请求入口
   */
  private static async request<T>(
    method: http.RequestMethod,
    url: string,
    data?: any
  ): Promise<T> {
    // 1. 拼接完整URL
    const fullUrl = this.BASE_URL + url;
    // 2. 构建请求配置
    const options: HttpRequestOptions = {
      method: method,
      header: this.getHeaders(),
      connectTimeout: this.TIMEOUT,
      readTimeout: this.TIMEOUT,
      expectDataType: http.HttpDataType.OBJECT // 自动解析JSON
    };
    // 3. 处理请求数据
    if (method === http.RequestMethod.GET && data) {
      // GET请求参数拼到URL
      const queryString = new URLSearchParams(data).toString();
      url = `${fullUrl}?${queryString}`;
    } else if (data) {
      // POST/PUT等请求体
      options.extraData = JSON.stringify(data);
    }
    // 4. 创建请求对象（不可复用）
    const httpRequest = http.createHttp();
    try {
      // 5. 发起请求
      const response = await httpRequest.request(fullUrl, options);
      // 6. 响应处理
      return this.handleResponse<T>(response);
    } catch (err) {
      // 7. 错误处理
      this.handleError(err);
      throw err;
    } finally {
      // 8. 销毁请求对象
      httpRequest.destroy();
    }
  }
  /**
   * 获取请求头（自动添加Token）
   */
  private static getHeaders(): Record<string, string> {
    const headers: Record<string, string> = {
      'Content-Type': 'application/json',
      'Accept': 'application/json'
    };
    // 从AppStorage获取Token并添加
    const token = AppStorage.get<string>('token');
    if (token) {
      headers['Authorization'] = `Bearer ${token}`;
    }
    return headers;
  }
  /**
   * 响应处理
   */
  private static handleResponse<T>(response: HttpResponse): T {
    const { responseCode, result } = response;
    console.info(`请求成功: ${responseCode}, 数据: ${JSON.stringify(result)}`);
    // 状态码判断
    if (responseCode >= 200 && responseCode < 300) {
      return result as T;
    } else {
      throw new Error(`请求失败，状态码: ${responseCode}`);
    }
  }
  /**
   * 错误处理
   */
  private static handleError(err: Error): void {
    console.error(`网络请求错误: ${err.message}`);
    // 可统一弹出Toast提示
    // promptAction.showToast({ message: '网络异常，请稍后重试' });
  }}
