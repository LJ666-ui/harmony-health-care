## 一、可视化需求分析

### 1.1 数据类型

| 数据类型 | 可视化方式 | 应用场景 |
|---------|-----------|---------|
| **生命体征** | 折线图、仪表盘 | 血压、心率、体温监测 |
| **健康指标** | 柱状图、雷达图 | 体检报告、健康评分 |
| **历史趋势** | 折线图、面积图 | 长期健康趋势分析 |
| **风险评估** | 仪表盘、进度条 | 疾病风险预警 |
| **对比分析** | 对比柱状图 | 不同时期数据对比 |

### 1.2 设计原则

1. **清晰性**: 数据展示清晰易懂
2. **准确性**: 数据精确无误
3. **实时性**: 支持动态更新
4. **交互性**: 支持用户交互
5. **性能**: 渲染时间<100ms

### 2.1 折线图组件

```typescript
/**
 * 健康数据折线图
 */
@Component
export struct HealthLineChart {
  @Prop data: ChartDataPoint[] = [];
  @Prop title: string = '';
  @Prop unit: string = '';
  @Prop showGrid: boolean = true;
  @StorageLink('isOldModeEnabled') isElderMode: boolean = false;

  build() {
    Column() {
      // 标题
      Text(this.title)
        .fontSize(this.isElderMode ? 20 : 16)
        .fontWeight(FontWeight.Bold)
        .margin({ bottom: 12 })

      // 图表区域
      Stack() {
        // 网格背景
        if (this.showGrid) {
          this.buildGrid()
        }

        // 折线
        this.buildLine()

        // 数据点
        this.buildDataPoints()
      }
      .width('100%')
      .height(200)

      // 图例
      this.buildLegend()
    }
    .padding(16)
    .backgroundColor('#FFFFFF')
    .borderRadius(12)
  }

  @Builder buildLine(): void {
    // 使用Canvas绘制折线
    Canvas(this.context)
      .width('100%')
      .height(200)
      .onReady(() => {
        this.drawLine();
      })
  }

  private drawLine(): void {
    const canvas = this.context;
    const paint = new Paint();

    // 设置画笔
    paint.setStrokeWidth(2);
    paint.setColor('#1677FF');
    paint.setStyle(PaintStyle.STROKE);

    // 绘制路径
    const path = new Path();
    this.data.forEach((point, index) => {
      const x = this.getXPosition(index);
      const y = this.getYPosition(point.value);

      if (index === 0) {
        path.moveTo(x, y);
      } else {
        path.lineTo(x, y);
      }
    });

    canvas.drawPath(path, paint);
  }
}
```

### 2.2 仪表盘组件

```typescript
/**
 * 健康指标仪表盘
 */
@Component
export struct HealthGauge {
  @Prop value: number = 0;
  @Prop maxValue: number = 100;
  @Prop title: string = '';
  @Prop unit: string = '';
  @Prop warningThreshold: number = 70;
  @Prop dangerThreshold: number = 90;

  build() {
    Column() {
      // 标题
      Text(this.title)
        .fontSize(16)
        .fontWeight(FontWeight.Bold)

      // 仪表盘
      Stack() {
        // 背景圆环
        this.buildBackgroundRing()

        // 进度圆环
        this.buildProgressRing()

        // 中心数值
        this.buildCenterValue()
      }
      .width(200)
      .height(200)

      // 状态说明
      this.buildStatusText()
    }
  }

  @Builder buildProgressRing(): void {
    Canvas(this.context)
      .width(200)
      .height(200)
      .onReady(() => {
        this.drawProgressRing();
      })
  }

  private drawProgressRing(): void {
    const canvas = this.context;
    const paint = new Paint();

    // 计算角度
    const percentage = this.value / this.maxValue;
    const angle = percentage * 360;

    // 设置颜色
    const color = this.getStatusColor();
    paint.setColor(color);
    paint.setStrokeWidth(12);
    paint.setStyle(PaintStyle.STROKE);
    paint.setStrokeCap(Cap.ROUND);

    // 绘制圆弧
    const rect = new Rect(10, 10, 180, 180);
    canvas.drawArc(rect, -90, angle, false, paint);
  }

  private getStatusColor(): string {
    if (this.value >= this.dangerThreshold) {
      return '#FF4D4F';  // 危险：红色
    } else if (this.value >= this.warningThreshold) {
      return '#FAAD14';  // 警告：黄色
    } else {
      return '#52C41A';  // 正常：绿色
    }
  }
}
```

### 2.3 雷达图组件

```typescript
/**
 * 健康评分雷达图
 */
@Component
export struct HealthRadarChart {
  @Prop dimensions: RadarDimension[] = [];
  @Prop title: string = '';

  build() {
    Column() {
      Text(this.title)
        .fontSize(16)
        .fontWeight(FontWeight.Bold)
        .margin({ bottom: 12 })

      Stack() {
        // 背景网格
        this.buildRadarGrid()

        // 数据区域
        this.buildRadarData()
      }
      .width(250)
      .height(250)

      // 维度标签
      this.buildDimensionLabels()
    }
  }

  @Builder buildRadarData(): void {
    Canvas(this.context)
      .width(250)
      .height(250)
      .onReady(() => {
        this.drawRadarData();
      })
  }

  private drawRadarData(): void {
    const canvas = this.context;
    const paint = new Paint();
    const path = new Path();

    const centerX = 125;
    const centerY = 125;
    const maxRadius = 100;

    // 绘制数据多边形
    this.dimensions.forEach((dim, index) => {
      const angle = (index * 2 * Math.PI) / this.dimensions.length - Math.PI / 2;
      const radius = (dim.value / dim.maxValue) * maxRadius;
      const x = centerX + radius * Math.cos(angle);
      const y = centerY + radius * Math.sin(angle);

      if (index === 0) {
        path.moveTo(x, y);
      } else {
        path.lineTo(x, y);
      }
    });

    path.close();

    // 填充
    paint.setColor('#1677FF20');
    paint.setStyle(PaintStyle.FILL);
    canvas.drawPath(path, paint);

    // 边框
    paint.setColor('#1677FF');
    paint.setStyle(PaintStyle.STROKE);
    paint.setStrokeWidth(2);
    canvas.drawPath(path, paint);
  }
}
```

```typescript
 * 图表数据点

interface ChartDataPoint {
  timestamp: number;    // 时间戳

/**
 * 图表配置
```

## 四、性能优化

### 4.1 渲染优化

```typescript
/**
 * 图表渲染优化
 */
export class ChartRenderOptimizer {
  /**
   * 数据采样
   * 减少数据点数量
   */
  static sampleData(data: ChartDataPoint[], maxPoints: number): ChartDataPoint[] {
    if (data.length <= maxPoints) {
      return data;
    }

    const step = Math.ceil(data.length / maxPoints);
    const sampled: ChartDataPoint[] = [];

    for (let i = 0; i < data.length; i += step) {
      sampled.push(data[i]);
    }

    return sampled;
  }

  /**
   * 虚拟化渲染
   * 只渲染可见区域
   */
  static virtualizeRender(
    data: ChartDataPoint[],
    viewport: Viewport
  ): ChartDataPoint[] {
    return data.filter(point =>
      point.timestamp >= viewport.start &&
      point.timestamp <= viewport.end
    );
  }
}
```
