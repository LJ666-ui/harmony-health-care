## 二、启动优化

### 2.1 延迟加载

```typescript
/**
 * 延迟加载策略
 */
@Entry
@Component
struct MainPage {
  @State isReady: boolean = false;

  aboutToAppear(): void {
    // 1. 加载关键数据
    this.loadCriticalData();

    // 2. 延迟加载非关键数据
    setTimeout(() => {
      this.loadNonCriticalData();
      this.isReady = true;
    }, 100);
  }

  build() {
    Column() {
      // 关键内容立即显示
      this.buildCriticalContent()

      // 非关键内容延迟显示
      if (this.isReady) {
        this.buildNonCriticalContent()
      } else {
        this.buildSkeleton()
      }
    }
  }
}
```

### 3.2 LazyForEach优化

```typescript
/**
 * 大数据列表优化
 */
@Component
struct OptimizedList {
  @State dataSource: LazyDataSource = new LazyDataSource();

  build() {
    List() {
      // ✅ 使用LazyForEach
      LazyForEach(
        this.dataSource,
        (item: DataItem) => {
          ListItem() {
            this.buildItem(item)
          }
        },
        (item: DataItem) => item.id  // 唯一键
      )
    }
    .width('100%')
    .height('100%')
  }

  @Builder buildItem(item: DataItem): void {
    // 使用@ComponentV2优化列表项
    ListItemComponent({ item: item })
  }
}
```

## 四、内存优化

### 4.1 图片缓存

```typescript
/**
 * 图片缓存管理
 */
export class ImageCacheManager {
  private cache: Map<string, ImageBitmap> = new Map();
  private maxSize: number = 50;  // 最大缓存数量

  /**
   * 获取图片
   */
  async getImage(url: string): Promise<ImageBitmap> {
    // 检查缓存
    let image = this.cache.get(url);
    if (image) {
      return image;
    }

    // 加载图片
    image = await this.loadImage(url);

    // 缓存图片
    if (this.cache.size >= this.maxSize) {
      // LRU淘汰
      this.evictLRU();
    }
    this.cache.set(url, image);

    return image;
  }

  /**
   * 及时释放
   */
  release(url: string): void {
    const image = this.cache.get(url);
    if (image) {
      image.release();
      this.cache.delete(url);
    }
  }
}
```

### 4.2 对象池

```typescript
/**
 * 对象池优化
 */
export class ObjectPool<T> {
  private pool: T[] = [];
  private factory: () => T;

  constructor(factory: () => T, initialSize: number = 10) {
    this.factory = factory;
    for (let i = 0; i < initialSize; i++) {
      this.pool.push(factory());
    }
  }

  /**
   * 获取对象
   */
  acquire(): T {
    if (this.pool.length > 0) {
      return this.pool.pop()!;
    }
    return this.factory();
  }

  /**
   * 释放对象
   */
  release(obj: T): void {
    this.pool.push(obj);
  }
}
```

## 五、网络优化

### 5.1 请求合并

```typescript
/**
 * 请求合并器
 */
export class RequestBatcher {
  private pendingRequests: Map<string, Promise<any>> = new Map();

  /**
   * 批量请求
   */
  async batchRequest<T>(key: string, request: () => Promise<T>): Promise<T> {
    // 检查是否有相同的待处理请求
    if (this.pendingRequests.has(key)) {
      return this.pendingRequests.get(key) as Promise<T>;
    }

    // 发起新请求
    const promise = request();
    this.pendingRequests.set(key, promise);

    try {
      const result = await promise;
      return result;
    } finally {
      this.pendingRequests.delete(key);
    }
  }
}
```

```typescript
export class DataPrefetcher {
  /**
   * 预取下一页数据
   */
  async prefetchNextPage(currentPage: number): Promise<void> {
    const nextPage = currentPage + 1;

    // 后台预取

      const data = await this.fetchPage(nextPage);
```
