# 无障碍友好界面设计规范

## 版本信息
- **项目**: 星云医疗助手
- **标准**: WCAG 2.1 AA级
- **目标用户**: 老年用户、视障用户、运动障碍用户、认知障碍用户

---

## 一、无障碍设计原则

### 1.1 WCAG 2.1四大原则

| 原则 | 说明 | 医疗应用重点 |
|------|------|-------------|
| **可感知** | 信息必须可被用户感知 | 大字体、高对比度、语音播报 |
| **可操作** | 界面组件必须可操作 | 大触控区域、简化操作、语音控制 |
| **可理解** | 信息和操作必须可理解 | 简洁文案、清晰导航、错误提示 |
| **健壮性** | 内容必须足够健壮 | 兼容辅助技术、标准化接口 |

### 1.2 医疗应用特殊需求

| 用户群体 | 特殊需求 | 设计重点 |
|---------|---------|---------|
| **老年用户** | 视力下降、操作困难 | 老年模式、大字体、简化流程 |
| **视障用户** | 无法看到界面 | 屏幕阅读器、语音导航 |
| **运动障碍** | 操作不便 | 语音控制、简化手势 |
| **认知障碍** | 理解困难 | 简洁文案、图标辅助、分步引导 |

  static readonly STANDARD = {
    primary: '#1677FF',      // 主色
    primaryText: '#333333',  // 主文字
    secondaryText: '#666666',// 次要文字
    disabledText: '#999999', // 禁用文字
    background: '#FFFFFF',   // 背景
    border: '#E0E0E0'        // 边框
  };

  // 高对比度模式
  static readonly HIGH_CONTRAST = {
    primary: '#0052CC',      // 更深的蓝色
    primaryText: '#000000',  // 纯黑文字
    secondaryText: '#333333',// 深灰文字
    disabledText: '#666666', // 中灰文字
    background: '#FFFFFF',   // 白色背景
    border: '#000000'        // 黑色边框
  };

  // 验证对比度
  static verifyContrastRatio(foreground: string, background: string): number {
    // 计算对比度
    const fgLuminance = this.getLuminance(foreground);
    const bgLuminance = this.getLuminance(background);
    const ratio = (Math.max(fgLuminance, bgLuminance) + 0.05) /
                  (Math.min(fgLuminance, bgLuminance) + 0.05);
    return ratio;
  }
}

/**
 * 语音控制接口

export class VoiceControl {
  /**
   * 语音命令处理
   */
  async handleVoiceCommand(command: string): Promise<void> {
    const intent = await this.recognizeIntent(command);

    switch (intent) {
      case 'appointment':
        await this.navigateToAppointment();
        break;
      case 'query_record':
        await this.showMedicalRecord();
        break;
      case 'call_doctor':
        await this.callDoctor();
        break;
      default:
        await this.speak('抱歉，我没有理解您的指令，请再说一次');
    }
  }

  /**
   * 语音播报
   */
  async speak(text: string): Promise<void> {
    // 使用TTS播报
    const tts = textToSpeech.createTextToSpeechEngine();
    await tts.speak(text, {
      speed: this.isElderMode ? 0.8 : 1.0,  // 老年模式慢速
      pitch: 1.0
    });
  }
}

#### 4.1.1 简化文案

```typescript
/**
 * 文案简化规则
 */
export class TextSimplification {
  // 医疗术语简化
  static readonly TERM_SIMPLIFICATION = {
    '高血压': '血压高',
    '糖尿病': '血糖高',
    '冠状动脉粥样硬化性心脏病': '冠心病',
    '上呼吸道感染': '感冒'
  };

  /**
   * 简化文本
   */
  static simplify(text: string, isElderly: boolean): string {
    if (!isElderly) {
      return text;  // 标准模式不简化
    }

    let simplified = text;
    for (const [complex, simple] of Object.entries(this.TERM_SIMPLIFICATION)) {
      simplified = simplified.replace(new RegExp(complex, 'g'), simple);
    }

    return simplified;
  }
}
```

@Component
struct AccessibleNavigation {
  @StorageLink('isOldModeEnabled') isElderMode: boolean = false;
  @State currentStep: number = 1;
  @State totalSteps: number = 3;

  build() {
    Column() {
      // 步骤指示器
      if (this.isElderMode) {
        this.buildStepIndicator()
      }

      // 内容区域
      this.buildContent()

      // 导航按钮
      this.buildNavigationButtons()
    }
  }

  @Builder buildStepIndicator(): void {
    Row() {
      Text(`第${this.currentStep}步，共${this.totalSteps}步`)
        .fontSize(20)
        .fontColor('#1677FF')
        .fontWeight(FontWeight.Bold)
    }
    .width('100%')
    .padding(16)
    .backgroundColor('#F0F7FF')
  }

  @Builder buildNavigationButtons(): void {
    Row() {
      if (this.currentStep > 1) {
        Button('上一步')
          .width('45%')
          .height(this.isElderMode ? 56 : 44)
          .onClick(() => {
            this.currentStep--;
          })
      }

      Button(this.currentStep === this.totalSteps ? '完成' : '下一步')
        .width('45%')
        .height(this.isElderMode ? 56 : 44)
        .onClick(() => {
          if (this.currentStep < this.totalSteps) {
            this.currentStep++;
          } else {
            this.complete();
          }
        })
    }
    .width('100%')
    .justifyContent(FlexAlign.SpaceEvenly)
    .padding(16)
  }
}
```

/**
 * 错误提示设计
 */
export class FriendlyErrorHandler {
  /**
   * 显示错误提示
   */
  showError(error: Error, isElderly: boolean): void {
    let message: string;

    // 根据用户类型定制错误信息
    if (isElderly) {
      message = this.getElderlyErrorMessage(error);
    } else {
      message = this.getStandardErrorMessage(error);
    }

    // 显示提示
    AlertDialog.show({
      title: '提示',
      message: message,
      autoCancel: true,
      confirm: {
        value: '我知道了',
        action: () => {}
      }
    });

    // 语音播报（老年模式）
    if (isElderly) {
      this.speak(message);
    }
  }

  /**
   * 老年用户错误信息
   */
  private getElderlyErrorMessage(error: Error): string {
    const errorMap: Record<string, string> = {
      'network': '网络不好，请稍后再试',
      'timeout': '操作超时，请重试',
      'not_found': '没有找到相关信息',
      'permission': '需要授权才能使用此功能'
    };

    return errorMap[error.type] || '操作失败，请重试';
  }
}
```

### 6.2 操作撤销

```typescript
/**
 * 操作撤销机制
 */
export class UndoableOperation {
  private operationHistory: Operation[] = [];

  /**
   * 执行可撤销操作
   */
  async execute(operation: Operation): Promise<void> {
    // 执行操作
    await operation.execute();

    // 保存到历史
    this.operationHistory.push(operation);

    // 显示撤销提示
    this.showUndoTip(operation);
  }

  /**
   * 显示撤销提示
   */
  private showUndoTip(operation: Operation): void {
    PromptAction.showToast({
      message: `${operation.name}成功，点击撤销`,
      duration: 3000,
      bottom: 100
    });
  }

  /**
   * 撤销操作
   */
  async undo(): Promise<void> {
    const lastOperation = this.operationHistory.pop();
    if (lastOperation) {
      await lastOperation.undo();
    }
  }
}
```

@Component
struct KeyboardNavigation {
  @State focusedIndex: number = 0;
  private items: string[] = ['预约挂号', '病历档案', '健康报告'];

  build() {
    Column() {
      ForEach(this.items, (item: string, index: number) => {
        Button(item)
          .width('100%')
          .height(44)
          .backgroundColor(this.focusedIndex === index ? '#E6F7FF' : '#FFFFFF')
          .border({
            width: this.focusedIndex === index ? 2 : 1,
            color: this.focusedIndex === index ? '#1677FF' : '#E0E0E0'
          })
          .onKeyEvent((event: KeyEvent) => {
            if (event.keyCode === KeyCode.KEYCODE_DPAD_DOWN) {
              // 下箭头：移动焦点
              this.focusedIndex = Math.min(this.focusedIndex + 1, this.items.length - 1);
            } else if (event.keyCode === KeyCode.KEYCODE_DPAD_UP) {
              // 上箭头：移动焦点
              this.focusedIndex = Math.max(this.focusedIndex - 1, 0);
            } else if (event.keyCode === KeyCode.KEYCODE_ENTER) {
              // 回车：执行操作
              this.handleItemSelect(index);
            }
          })
      })
    }
  }
}

export class AccessibilityAutomatedTest {
  /**
   * 运行测试
   */
  async runTests(): Promise<TestResult[]> {
    const results: TestResult[] = [];

    // 1. 对比度测试
    results.push(await this.testContrastRatio());

    // 2. 触控区域测试

    // 3. 无障碍标签测试
    results.push(await this.testAccessibilityLabels());

    // 4. 键盘导航测试
    results.push(await this.testKeyboardNavigation());

    return results;
  }

  /**
   * 对比度测试
   */
  private async testContrastRatio(): Promise<TestResult> {
    const elements = this.getAllTextElements();
    const failures: string[] = [];

    for (const element of elements) {
      const ratio = this.calculateContrastRatio(element);
      if (ratio < 4.5) {
        failures.push(`元素${element.id}对比度${ratio}不达标`);
      }
    }

    return {
      name: '对比度测试',
      passed: failures.length === 0,
      failures: failures
    };
  }
}

## 九、最佳实践总结

### 9.1 设计原则

1. **以用户为中心**: 了解不同用户群体的需求
2. **渐进增强**: 基础功能优先，高级功能可选
3. **一致性**: 保持界面和交互的一致性
4. **容错性**: 提供友好的错误处理和撤销机制

### 9.2 实施建议

1. **早期集成**: 在设计阶段就考虑无障碍
2. **持续测试**: 定期进行无障碍测试
3. **用户反馈**: 收集真实用户的反馈
4. **文档完善**: 提供无障碍使用说明
