### 2.2 测试示例

```typescript
/**
 * HealthUtils单元测试
 */
import { describe, it, expect, beforeEach } from '@ohos/hypium';
import { HealthUtils } from '../../../utils/HealthUtils';

describe('HealthUtils', () => {
  describe('calculateBMI', () => {
    it('should calculate BMI correctly', () => {
      const weight = 70;  // kg
      const height = 1.75;  // m
      const bmi = HealthUtils.calculateBMI(weight, height);

      expect(bmi).toBeCloseTo(22.86, 2);
    });

    it('should throw error for invalid input', () => {
      expect(() => {
        HealthUtils.calculateBMI(-70, 1.75);
      }).toThrow();

      expect(() => {
        HealthUtils.calculateBMI(70, 0);
      }).toThrow();
    });
  });

  describe('assessHealthStatus', () => {
    it('should return normal for BMI 18.5-24.9', () => {
      const status = HealthUtils.assessHealthStatus(22);
      expect(status).toBe('normal');
    });

    it('should return overweight for BMI 25-29.9', () => {
      const status = HealthUtils.assessHealthStatus(27);
      expect(status).toBe('overweight');
    });
  });
});
```

## 三、集成测试

### 3.1 API集成测试

```typescript
/**
 * 医院服务集成测试
 */
import { describe, it, expect, beforeAll, afterAll } from '@ohos/hypium';
import { HospitalService } from '../../../service/HospitalService';

describe('HospitalService Integration', () => {
  let service: HospitalService;

  beforeAll(() => {
    service = new HospitalService();
  });

  it('should fetch hospital list from API', async () => {
    const hospitals = await service.getHospitalList();

    expect(hospitals).toBeDefined();
    expect(hospitals.length).toBeGreaterThan(0);
    expect(hospitals[0]).toHaveProperty('id');
    expect(hospitals[0]).toHaveProperty('name');
  });

  it('should fetch hospital detail by ID', async () => {
    const hospitalId = '1';
    const detail = await service.getHospitalDetail(hospitalId);

    expect(detail).toBeDefined();
    expect(detail.id).toBe(hospitalId);
  });
});
```

## 四、UI测试

### 4.1 关键流程测试

```typescript
/**
 * 预约流程UI测试
 */
import { Driver, Component } from '@ohos.aceTestKit';

describe('Appointment Flow', () => {
  let driver: Driver;

  beforeAll(async () => {
    driver = Driver.create();
    await driver.start();
  });

  it('should complete appointment flow', async () => {
    // 1. 打开首页
    await driver.launchApp();

    // 2. 点击预约挂号
    const appointmentBtn = await driver.findComponent(
      By.text('预约挂号')
    );
    await appointmentBtn.click();

    // 3. 选择医院
    const hospital = await driver.findComponent(
      By.text('北京协和医院')
    );
    await hospital.click();

    // 4. 选择科室
    const department = await driver.findComponent(
      By.text('内科')
    );
    await department.click();

    // 5. 验证到达预约页面
    const pageTitle = await driver.findComponent(
      By.type('Text')
    );
    const text = await pageTitle.getText();
    expect(text).toContain('预约');
  });

  afterAll(async () => {
    await driver.stop();
  });
});
```

## 五、性能测试

### 5.1 启动性能测试

```typescript
/**
 * 启动性能测试
 */
import { HiPerf } from '@ohos/hiPerf';

describe('Startup Performance', () => {
  it('should start within 2 seconds', async () => {
    const startTime = Date.now();

    // 启动应用
    await driver.launchApp();

    // 等待首页加载完成
    await driver.findComponent(By.type('Column'));

    const endTime = Date.now();
    const duration = endTime - startTime;

    expect(duration).toBeLessThan(2000);
  });
});
```

### 5.2 滚动性能测试

```typescript
/**
 * 滚动性能测试
 */
describe('Scroll Performance', () => {
  it('should maintain 55fps during scroll', async () => {
    // 启动性能监控
    HiPerf.startMonitor('fps');

    // 执行滚动
    const list = await driver.findComponent(By.type('List'));
    await list.scroll(1000);  // 滚动1000像素

    // 获取FPS数据
    const fpsData = HiPerf.getMonitorData('fps');
    const avgFps = fpsData.reduce((a, b) => a + b, 0) / fpsData.length;

    expect(avgFps).toBeGreaterThan(55);

    HiPerf.stopMonitor('fps');
  });
});
```

## 六、CI/CD集成

### 6.1 测试脚本

```yaml
# .github/workflows/test.yml
name: Test

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2

      - name: Setup HarmonyOS SDK
        uses: harmonyos/setup-sdk@v1

      - name: Run Unit Tests
        run: |
          hvigorw test --type unit

      - name: Run Integration Tests
        run: |
          hvigorw test --type integration

      - name: Generate Coverage Report
        run: |
          hvigorw coverage

      - name: Upload Coverage
        uses: codecov/codecov-action@v2
```
