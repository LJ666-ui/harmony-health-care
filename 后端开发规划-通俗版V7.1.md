```
爬虫脚本结构:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# herbal_crawler.py
import requests
from bs4 import BeautifulSoup
import json
import pymysql
import time
import random

# 配置
BASE_URL = "https://www.zhongyoo.com"
DB_CONFIG = {
    "host": "localhost",
    "user": "root",
    "password": "your_password",
    "database": "medical_health"
}

# 目标分类URL映射
CATEGORY_URLS = {
    "补气药": "/yaocai/buqi/",
    "补血药": "/yaocai/buxue/",
    "补阴药": "/yaocai/buyin/",
    "补阳药": "/yaocai/buyang/",
    "清热药": "/yaocai/qingre/",
    "解表药": "/yaocai/jiebiao/",
    # ... 更多分类
}

def crawl_category(category_name, url):
    """爬取某个分类下的所有药材"""
    herbs = []
    response = requests.get(BASE_URL + url, headers={"User-Agent": "Mozilla/5.0"})
    soup = BeautifulSoup(response.text, 'html.parser')

    for item in soup.select('.herb-item'):
        herb = {
            "name": item.select_one('.name').text.strip(),
            "pinyin": item.select_one('.pinyin').text.strip(),
            "alias": item.select_one('.alias').text.strip() if item.select_one('.alias') else "",
            "category": category_name,
            "nature": item.select_one('.nature').text.strip(),
            "taste": item.select_one('.taste').text.strip(),
            "meridian": item.select_one('.meridian').text.strip(),
            "efficacy_summary": item.select_one('.efficacy').text.strip(),
            # ... 更多字段
        }
        herbs.append(herb)
        time.sleep(random.uniform(0.5, 1.5))  # 礼貌爬取

    return herbs

def save_to_db(herbs):
    """保存到MySQL"""
    conn = pymysql.connect(**DB_CONFIG)
    cursor = conn.cursor()
    sql = """
        INSERT INTO herbal_medicine
        (name, pinyin, alias, category, nature, taste, meridian,
         efficacy_summary, toxicity, create_time)
        VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s, NOW())
    """
    for herb in herbs:
        cursor.execute(sql, (
            herb['name'], herb['pinyin'], herb['alias'],
            herb['category'], herb['nature'], herb['taste'],
            herb['meridian'], herb['efficacy_summary'], herb['toxicity']
        ))
    conn.commit()
    print(f"✅ 成功导入 {len(herbs)} 条药材数据")

def main():
    all_herbs = []
    for cat_name, cat_url in CATEGORY_URLS.items():
        print(f"📦 正在爬取: {cat_name}")
        herbs = crawl_category(cat_name, cat_url)
        all_herbs.extend(herbs)
        print(f"   → 获取 {len(herbs)} 条")

    save_to_db(all_herbs)
    print(f"\n🎉 总计导入 {len(all_herbs)} 种中药材!")

if __name__ == "__main__":
    main()

运行方式:

```

```sql
CREATE TABLE `knowledge_edge` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT,
  `source_node_id` bigint(20) NOT NULL COMMENT '起始节点ID',
  `target_node_id` bigint(20) NOT NULL COMMENT '目标节点ID',

  `relation_name` varchar(100) DEFAULT NULL COMMENT '中文显示名',

  KEY `idx_target` (`target_node_id`),
```

```java
public PageResult<HerbalMedicine> search(HerbalSearchRequest request) {
    QueryWrapper<HerbalMedicine> wrapper = new QueryWrapper<>();

    if (StringUtils.isNotBlank(request.getKeyword())) {
        wrapper.and(w -> w
            .like("name", request.getKeyword())
            .or().like("pinyin", request.getKeyword())
            .or().like("alias", request.getKeyword())
        );
    }

    if (StringUtils.isNotBlank(request.getCategory())) {

    }

    Page<HerbalMedicine> page = new Page<>(request.getPage(), request.getSize());

}
```

### 6.2 知识图谱BFS子图查询

```java
// KnowledgeGraphServiceImpl.java 核心算法
public GraphData getSubGraph(Long centerId, Integer depth, Integer maxNodes) {
    Set<Long> visited = new HashSet<>();
    List<KnowledgeNode> nodes = new ArrayList<>();
    List<KnowledgeEdge> edges = new ArrayList<>();

    Queue<Long> queue = new LinkedList<>();
    queue.add(centerId);
    visited.add(centerId);
    int currentDepth = 0;

    while (!queue.isEmpty() && currentDepth < depth && nodes.size() < maxNodes) {
        int levelSize = queue.size();
        for (int i = 0; i < levelSize && nodes.size() < maxNodes; i++) {
            Long nodeId = queue.poll();
            KnowledgeNode node = nodeMapper.selectById(nodeId);
            if (node != null) {
                assignPosition(node, nodes.size());
                nodes.add(node);
            }

            List<KnowledgeEdge> outEdges = edgeMapper.selectList(
                new QueryWrapper<KnowledgeEdge>().eq("source_node_id", nodeId)
            );
            for (KnowledgeEdge edge : outEdges) {
                if (!visited.contains(edge.getTargetNodeId()) && nodes.size() < maxNodes) {
                    visited.add(edge.getTargetNodeId());
                    queue.add(edge.getTargetNodeId());
                }
                edges.add(edge);
            }
        }
        currentDepth++;
    }

    return new GraphData(nodes, edges);
}

private void assignPosition(KnowledgeNode node, int index) {
    double angle = 2 * Math.PI * index / 50;
    double radius = 150 + (index % 5) * 30;
    node.setX(180 + (int)(radius * Math.cos(angle)));
    node.setY(250 + (int)(radius * Math.sin(angle)));
}
```
