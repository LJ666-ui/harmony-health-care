## 一、项目概述与当前进展

### 1.1 项目定位

**星云医疗助手**是一款基于HarmonyOS原生开发的分布式智慧医疗全周期管理应用，聚焦老年群体就医、康复、健康管理的核心痛点。

### 1.2 当前进展

| 阶段 | 状态 | 完成内容 |
|------|------|----------|
| Day 1-2 | ✅ 完成 | 环境搭建、项目结构、基础组件库 |
| Day 3 | ✅ 完成 | Index.ets、Login.ets 页面开发 |
| Day 4 | 🔄 部分 | HomePage.ets 页面框架搭建 |
| Day 5-7 | ⏳ 待开发 | 健康记录、预约管理、用药管理、个人中心页面 |

### 1.3 技术栈

- **框架**：HarmonyOS SDK 6.0.2 (API 22)
- **语言**：ArkTS
- **UI框架**：ArkUI 9.0.0
- **路由**：@kit.ArkUI router

```typescript
// 位置：entry/src/main/ets/components/Footer.ets
import { router } from '@kit.ArkUI';

@Component
export struct Footer {
  // 当前选中的索引
  @State selectedIndex: number = 0;

  private tabs: string[] = ['首页', '健康', '预约', '我的'];

  build() {
    Row() {
      ForEach(this.tabs, (tab: string, index: number) => {
        Column() {
          Text(tab)
            .fontSize(index === this.selectedIndex ? 16 : 14)
            .fontColor(index === this.selectedIndex ? '#1677FF' : '#999999')
        }
        .flexGrow(1)
        .onClick(() => {
          this.selectedIndex = index;
          this.navigateToPage(index);
        })
      })
    }
    .width('100%')
    .height(60)
    .backgroundColor('#FFFFFF')
    .border({
      width: { top: 1 },
      color: '#E5E5E5'
    })
  }

  private navigateToPage(index: number): void {
    const pages: string[] = [
      'pages/HomePage',
      'pages/HealthRecords',
      'pages/Appointments',
      'pages/Profile'
    ];
    router.pushUrl({ url: pages[index] });
  }
}
```

```typescript
          label: '密码',
          placeholder: '请输入密码',
          isPassword: true,
          value: this.password
        })
        .margin({ bottom: 30 })
        
        // 登录按钮
        Button('登录')
          .width('100%')
          .height(48)
          .fontSize(18)
          .fontWeight(600)
          .fontColor('#FFFFFF')
          .backgroundColor(this.isLoading ? '#99CCFF' : '#1677FF')
          .borderRadius(24)
          .onClick(() => {
            this.handleLogin();
          })
        
        // 注册链接
        Row() {
          Text('还没有账号？')
            .fontSize(14)
            .fontColor('#666666')
          Text('立即注册')
            .fontSize(14)
            .fontColor('#1677FF')
            .onClick(() => {
              router.pushUrl({ url: 'pages/Register' });
            })
        }
        .margin({ top: 20 })
      }
      .flexGrow(1)
      .width('100%')
      .padding({ left: 32, right: 32, top: 40 })
    }
    .width('100%')
    .height('100%')
    .backgroundColor('#FFFFFF')
  }

  handleLogin(): void {
    if (!this.username || !this.password) {
      promptAction.showToast({ message: '请输入用户名和密码' });
      return;
    }
    
    this.isLoading = true;
    
    // TODO: 调用登录API
    setTimeout(() => {
      this.isLoading = false;
      router.pushUrl({
        url: 'pages/HomePage'
      });
    }, 1000);
  }
}
```

```typescript

import { Footer } from '../components/Footer';

@Entry
@Component
struct HealthRecords {
  @State selectedTab: number = 0;
  @State healthDataList: HealthData[] = [];
  
  private tabs: string[] = ['全部', '血压', '血糖', '心率', '体重'];

  aboutToAppear(): void {
    this.loadMockData();
  }

  loadMockData(): void {
    this.healthDataList = [
      { id: '1', type: 'blood_pressure', value: '120/80', time: '2026-04-09 10:00', status: 'normal' },
      { id: '2', type: 'blood_sugar', value: '5.6', time: '2026-04-09 08:00', status: 'normal' },
      { id: '3', type: 'heart_rate', value: '72', time: '2026-04-08 20:00', status: 'normal' },
      { id: '4', type: 'weight', value: '65.5', time: '2026-04-08 08:00', status: 'normal' }
    ];
  }

  build() {
    Column() {
      Header({ title: '健康记录', showBack: true })
      
      // Tab选择器
      Row() {
        ForEach(this.tabs, (tab: string, index: number) => {
          Column() {
            Text(tab)
              .fontSize(index === this.selectedTab ? 16 : 14)
              .fontColor(index === this.selectedTab ? '#1677FF' : '#666666')
              .fontWeight(index === this.selectedTab ? 600 : 400)
            
            if (index === this.selectedTab) {
              Divider()
                .width(24)
                .height(3)
                .backgroundColor('#1677FF')
                .borderRadius(2)
                .margin({ top: 4 })
            }
          }
          .flexGrow(1)
          .onClick(() => {
            this.selectedTab = index;
          })
        })
      }
      .width('100%')
      .padding({ top: 12, bottom: 12 })
      .backgroundColor('#FFFFFF')

      // 数据列表
      Scroll() {
        Column() {
          // 添加按钮
          Button('+ 添加健康记录')
            .width('100%')
            .height(48)
            .fontSize(16)
            .fontColor('#1677FF')
            .fontWeight(600)
            .backgroundColor('#E6F7FF')
            .borderRadius(24)
            .onClick(() => {
              this.showAddDialog();
            })
            .margin({ bottom: 16 })

          // 数据列表
          ForEach(this.healthDataList, (item: HealthData) => {
            this.HealthDataItem(item)
          })
        }
        .width('100%')
        .padding({ left: 20, right: 20, top: 16, bottom: 20 })
      }
      .flexGrow(1)
      .scrollBar(BarState.Off)
      
      Footer()
    }
    .width('100%')
    .height('100%')
    .backgroundColor('#F5F7FA')
  }

  @Builder
  HealthDataItem(item: HealthData): void {
    Row() {
      Column() {
        Text(this.getTypeIcon(item.type))
          .fontSize(24)
      }
      .width(48)
      .height(48)
      .backgroundColor('#F5F7FA')
      .borderRadius(12)
      .justifyContent(FlexAlign.Center)
      
      Column() {
        Text(this.getTypeName(item.type))
          .fontSize(16)
          .fontWeight(600)
          .fontColor('#333333')
        
        Text(item.time)
          .fontSize(12)
          .fontColor('#999999')
          .margin({ top: 4 })
      }
      .alignItems(HorizontalAlign.Start)
      .margin({ left: 12 })
      
      Blank()
      
      Column() {
        Text(item.value)
          .fontSize(20)
          .fontWeight(600)
          .fontColor(item.status === 'normal' ? '#333333' : '#FF6B6B')
        
        Text(this.getTypeUnit(item.type))
          .fontSize(12)
          .fontColor('#999999')
      }
    }
    .width('100%')
    .padding(16)
    .backgroundColor('#FFFFFF')
    .borderRadius(12)
    .margin({ bottom: 12 })
  }
```
