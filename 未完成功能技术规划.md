```typescript
/**
 * 分布式跨院数据协同核心管理器
 * 职责：设备发现、配对认证、连接管理、数据对象创建
 */
export class DistributedManager {
  private static instance: DistributedManager;
  private deviceManager: distributedDeviceManager.DeviceManager;
  private kvStore: distributedKVStore.KVStore | null = null;
  private connectedDevices: Map<string, DeviceInfo> = new Map();

  private constructor() {}

  public static getInstance(): DistributedManager {
    if (!DistributedManager.instance) {
      DistributedManager.instance = new DistributedManager();
    }
    return DistributedManager.instance;
  }

  /**
   * 初始化分布式设备管理器
   */
  async initialize(): Promise<void> {
    this.deviceManager = distributedDeviceManager.createDeviceManager('com.example.harmonyhealthcare');
    
    // 注册设备状态变化监听
    this.deviceManager.on('deviceStateChange', (data) => {
      console.log('[Distributed] 设备状态变更: ' + JSON.stringify(data));
      this.handleDeviceStateChange(data);
    });
  }

  /**
   * 发现附近鸿蒙设备（医院终端、医生手机等）
   */
  async discoverDevices(timeoutMs: number = 10000): Promise<DeviceInfo[]> {
    const devices: DeviceInfo[] = [];
    
    const deviceList = this.deviceManager.getTrustedDeviceListSync();
    
    for (let i = 0; i < deviceList.length; i++) {
      const device = deviceList[i];
      devices.push({
        deviceId: device.deviceId,
        deviceName: device.deviceName,
        deviceType: this.getDeviceType(device.deviceType),
        isOnline: true,
        lastConnectTime: Date.now()
      });
    }
    
    return devices;
  }

  /**
   * 与目标设备建立配对连接
   */
  async pairDevice(deviceId: string, pinCode?: string): Promise<PairingResult> {
    try {
      // 发起配对请求
      const result = await this.deviceManager.authenticateDevice(deviceId, {
        extraInfo: pinCode ? { pinCode } : {}
      });

      if (result === 0) {
        // 配对成功，建立安全通道
        await this.establishSecureChannel(deviceId);
        
        return {
          success: true,
          message: '设备配对成功',
          deviceId: deviceId
        };
      } else {
        return {
          success: false,
          message: '设备配对失败，错误码: ' + result,
          errorCode: result
        };
      }
    } catch (error) {
      return {
        success: false,
        message: '配对异常: ' + error.message,
        errorCode: -1
      };
    }
  }

  /**
   * 创建分布式KVStore用于数据同步
   */
  async createDataStore(storeId: string): Promise<distributedKVStore.KVStore> {
    const config: distributedKVStore.Options = {
      createIfMissing: true,
      encrypt: true,  // 开启自动加密
      securityLevel: distributedKVStore.SecurityLevel.S1,  // 最高安全级别
      storeId: storeId
    };

    this.kvStore = await distributedKVStore.createKVStore(config);
    
    // 注册数据变更监听
    this.kvStore.on('dataChange', distributedKVStore.SubscribeType.SUBSCRIBE_LOCAL, (changeData) => {
      console.log('[Distributed] 本地数据变更');
      this.handleLocalDataChange(changeData);
    });

    this.kvStore.on('dataChange', distributedKVStore.SubscribeType.SUBSCRIBE_REMOTE, (changeData) => {
      console.log('[Distributed] 远程数据同步到达');
      this.handleRemoteDataSync(changeData);
    });

    return this.kvStore;
  }

  /**
   * 同步病历数据到目标医院设备
   */
  async syncMedicalRecord(
    recordId: string,
    targetDeviceId: string,
    permissionLevel: PermissionLevel
  ): Promise<SyncResult> {
    try {
      // 1. 从本地数据库读取病历
      const record = await this.getLocalMedicalRecord(recordId);
      
      // 2. 根据权限等级进行脱敏处理
      const desensitizedRecord = DesensitizationUtil.desensitize(record, permissionLevel);
      
      // 3. AES-256加密
      const encryptedData = await DesensitizationUtil.encryptData(
        JSON.stringify(desensitizedRecord),
        this.getSessionKey(targetDeviceId)
      );
      
      // 4. 写入分布式KVStore（自动同步到目标设备）
      await this.kvStore.put('medical_record_' + recordId + '_' + targetDeviceId, encryptedData);
      
      return {
        success: true,
        syncId: generateUUID(),
        timestamp: Date.now(),
        targetDevice: targetDeviceId
      };
    } catch (error) {
      return {
        success: false,
        error: error.message
      };
    }
  }
}
```

##### DesensitizationUtil.ets（隐私保护工具）
```typescript
/**
 * 敏感数据脱敏与AES-256加密工具类
 * 符合HIPAA/GDPR医疗数据隐私标准
 */
export class DesensitizationUtil {

  /**
   * 身份证号脱敏：110***********1234
   */
  static desensitizeIdCard(idCard: string): string {
    if (!idCard || idCard.length < 8) return idCard;
    return idCard.substring(0, 3) + '*'.repeat(idCard.length - 7) + idCard.substring(idCard.length - 4);
  }

  /**
   * 手机号脱敏：138****1234
   */
  static desensitizePhone(phone: string): string {
    if (!phone || phone.length < 7) return phone;
    return phone.substring(0, 3) + '****' + phone.substring(phone.length - 4);
  }

  /**
   * 姓名脱敏：张**
   */
  static desensitizeName(name: string): string {
    if (!name || name.length === 0) return name;
    if (name.length <= 2) return name.charAt(0) + '*';
    return name.charAt(0) + '*'.repeat(name.length - 1);
  }

  /**
   * 地址脱敏：北京市朝阳区***街道
   */
  static desensitizeAddress(address: string): string {
    if (!address) return address;
    const parts = address.split(/[省市区县]/);
    if (parts.length >= 2) {
      return parts[0] + '省' + (parts[1] || '') + '市***';
    }
    return address.substring(0, Math.min(6, address.length)) + '***';
  }

  /**
   * 病历内容智能脱敏（正则匹配 + 关键词识别）
   * @param content 原始病历内容
   * @param level 脱敏等级：FULL(完全)/PARTIAL(部分)/MINIMAL(最小)
   */
  static desensitizeMedicalRecord(content: string, level: DesensitizeLevel): string {
    let result = content;

    // 正则替换敏感信息
    const patterns = [
      { regex: /(\d{17}[\dXx]|\d{15})/g, replacement: '[身份证号]' },           // 身份证号
      { regex: /1[3-9]\d{9}/g, replacement: '[手机号码]' },                        // 手机号
      { regex: /[\w\.-]+@[\w\.-]+\.\w+/g, replacement: '[邮箱地址]' },             // 邮箱
      { regex: /(?:住址|地址)[：:]\s*[^\n]+/g, replacement: '[地址已隐藏]' },       // 地址字段
      { regex: /(姓名|患者)[：:]\s*[\u4e00-\u9fa5]{2,4}/g, replacement: '$1: ***' } // 姓名
    ];

    patterns.forEach(pattern => {
      result = result.replace(pattern.regex, pattern.replacement);
    });

    // 根据脱敏等级进一步处理
    switch (level) {
      case DesensitizeLevel.FULL:
        // 移除详细描述中的个人信息
        result = result.replace(/(现病史|既往史|个人史)[：:][\s\S]*?(?=\n[^\s]|$)/g, '$1: [已脱敏]');
        break;
      case DesensitizeLevel.PARTIAL:
        // 保留诊断结果，隐藏详细信息
        break;
      case DesensitizeLevel.MINIMAL:
        // 仅脱敏明确标识符
        break;
    }

    return result;
  }

  /**
   * AES-256-GCM加密（用于分布式传输）
   */
  static async encryptData(plainText: string, keyBase64: string): Promise<string> {
    try {
      let symKeyGenerator = cryptoFramework.createSymKeyGenerator('AES256');
      
      let keyData: cryptoFramework.DataBlob = { data: stringToUint8Array(atob(keyBase64)) };
      let symKey = await symKeyGenerator.convertKey(keyData);

      let cipher = cryptoFramework.createCipher('AES256|GCM|PKCS7');
      
      let iv = cryptoFramework.createRandomBytes(12);  // GCM推荐12字节IV
      
      await cipher.init(cryptoFramework.CryptoMode.ENCRYPT_MODE, symKey, { iv: iv });
      
      let plainBlob: cryptoFramework.DataBlob = { data: stringToUint8Array(plainText) };
      let encryptedBlob = await cipher.doFinal(plainBlob);

      let combined = new Uint8Array(iv.data.length + encryptedBlob.data.length);
      combined.set(iv.data, 0);
      combined.set(encryptedBlob.data, iv.data.length);
      
      return btoa(String.fromCharCode(...combined));
    } catch (error) {
      console.error('[Desensitization] 加密失败:', error.message);
      throw error;
    }
  }

  /**
   * AES-256-GCM解密
   */
  static async decryptData(encryptedBase64: string, keyBase64: string): Promise<string> {
    try {
      let symKeyGenerator = cryptoFramework.createSymKeyGenerator('AES256');
      let keyData: cryptoFramework.DataBlob = { data: stringToUint8Array(atob(keyBase64)) };
      let symKey = await symKeyGenerator.convertKey(keyData);

      let cipher = cryptoFramework.createCipher('AES256|GCM|PKCS7');
      
      let combined = stringToUint8Array(atob(encryptedBase64));
      let iv: cryptoFramework.DataBlob = { data: combined.slice(0, 12) };
      let ciphertext: cryptoFramework.DataBlob = { data: combined.slice(12) };

      await cipher.init(cryptoFramework.CryptoMode.DECRYPT_MODE, symKey, { iv: iv });
      
      let decryptedBlob = await cipher.doFinal(ciphertext);
      
      return uint8ArrayToString(decryptedBlob.data);
    } catch (error) {
      console.error('[Desensitization] 解密失败:', error.message);
      throw error;
    }
  }
}

export enum DesensitizeLevel {
  FULL = 'FULL',         // 完全脱敏（家属查看）
  PARTIAL = 'PARTIAL',   // 部分脱敏（其他医院医生）
  MINIMAL = 'MINIMAL'    // 最小脱敏（主治医生可见更多）
}
```

```typescript
/**
 * 转院审批页面（医生专用）
 * 功能：查看待审批申请 → 查看脱敏病历 → 审批通过/拒绝 → 触发数据同步
 */
@Entry
@Component
struct TransferApprovalPage {
  @StorageLink('isOldModeEnabled') isElderMode: boolean = false;

  @State pendingApplications: TransferApplication[] = [];
  @State selectedApplication: TransferApplication | null = null;
  @State patientRecords: MedicalRecord[] = [];
  @State isLoading: boolean = false;
  @State showDetailDialog: boolean = false;
  @State approvalComment: string = '';
  @State showSuccessToast: boolean = false;

  private distributedManager: DistributedManager = DistributedManager.getInstance();

  aboutToAppear(): void {
    this.loadPendingApplications();
  }

  async loadPendingApplications(): Promise<void> {
    this.isLoading = true;
    try {
      const response = await HttpUtil.get<TransferApplication[]>(
        ApiConstants.TRANSFER_PENDING_LIST
      );
      if (response.success && response.data) {
        this.pendingApplications = response.data;
      }
    } finally {
      this.isLoading = false;
    }
  }

  selectApplication(app: TransferApplication): void {
    this.selectedApplication = app;
    this.showDetailDialog = true;
    this.loadPatientRecords(app.patientId);
  }

  async loadPatientRecords(patientId: string): Promise<void> {
    const response = await HttpUtil.get<MedicalRecord[]>(
      ApiConstants.MEDICAL_RECORD_LIST + '?patientId=' + patientId
    );
    if (response.success && response.data) {
      this.patientRecords = response.data;
    }
  }

  async approveTransfer(isApproved: boolean): Promise<void> {
    if (!this.selectedApplication || !this.approvalComment.trim()) {
      ToastUtil.showToast('请填写审批意见');
      return;
    }

    this.isLoading = true;
    try {
      const response = await HttpUtil.post(ApiConstants.TRANSFER_APPROVE, {
        applicationId: this.selectedApplication.id,
        status: isApproved ? 'approved' : 'rejected',
        comment: this.approvalComment,
        reviewerId: AppStorage.get<string>('userId'),
        reviewTime: new Date().toISOString()
      });

      if (response.success) {
        if (isApproved) {
          await this.triggerDataSync(this.selectedApplication);
        }

        this.showSuccessToast = true;
        this.showDetailDialog = false;
        
        await this.loadPendingApplications();
      }
    } finally {
      this.isLoading = false;
    }
  }

  async triggerDataSync(application: TransferApplication): Promise<void> {
    try {
      const devices = await this.distributedManager.discoverDevices();
      const hospitalDevice = devices.find(d => 
        d.deviceType === DeviceType.HOSPITAL_TERMINAL && 
        d.deviceName.includes(application.targetHospital)
      );

      if (hospitalDevice) {
        const syncResult = await this.distributedManager.syncMedicalRecord(
          application.patientId,
          hospitalDevice.deviceId,
          PermissionLevel.DOCTOR
        );

        console.log('[Transfer] 数据同步结果:', JSON.stringify(syncResult));
      }
    } catch (error) {
      console.error('[Transfer] 数据同步失败:', error.message);
    }
  }

  build() {
    Column() {
      Header({ title: '转院审批', showBack: true });

      if (this.isLoading) {
        Loading();
      } else if (this.pendingApplications.length === 0) {
        EmptyState({ icon: '📋', message: '暂无待审批的转院申请' });
      } else {
        List({ space: 12 }) {
          ForEach(this.pendingApplications, (app: TransferApplication) => {
            ListItem() {
              Card({
                onClick: () => this.selectApplication(app)
              }) {
                Column({ space: 8 }) {
                  Row() {
                    Text('患者: ' + app.patientName)
                      .fontSize(16)
                      .fontWeight(FontWeight.Bold);
                    Blank();
                    Text(this.formatDate(app.applyTime))
                      .fontSize(12)
                      .fontColor('#999999');
                  }
                  .width('100%');

                  Text('原医院: ' + app.fromHospital)
                    .fontSize(14)
                    .fontColor('#666666');

                  Text('目标医院: ' + app.targetHospital)
                    .fontSize(14)
                    .fontColor('#1677FF');

                  Text('申请原因: ' + app.reason)
                    .fontSize(13)
                    .fontColor('#888888')
                    .maxLines(2)
                    .textOverflow({ overflow: TextOverflow.Ellipsis });

                  Row() {
                    Text('状态: 待审批')
                      .fontSize(12)
                      .fontColor('#FA8C16')
                      .backgroundColor('#FFF7E6')
                      .padding({ left: 8, right: 8, top: 4, bottom: 4 })
                      .borderRadius(4);

                    Blank();

                    Button('立即审批')
                      .type(ButtonType.Capsule)
                      .backgroundColor('#1677FF')
                      .fontSize(12)
                      .onClick(() => this.selectApplication(app));
                  }
                  .width('100%');
                }
                .alignItems(HorizontalAlign.Start)
                .padding(16);
              }
            }
          });
        }
        .layoutWeight(1)
        .padding(16);
      }

      Footer({ currentPage: 'transfer' });
    }
    .height('100%')
    .backgroundColor('#F5F5F5');
  }
}
```

```typescript
/**
 * 健康数据特征提取器 - 将原始数据转换为10维标准化特征向量
 * 特征维度:
 * [0]: 收缩压归一化 (60-180mmHg)
 * [1]: 舒张压归一化 (40-120mmHg)
 * [2]: 空腹血糖归一化 (3.9-15.0mmol/L)
 * [3]: BMI归一化 (15-40)
 * [4]: 年龄归一化 (18-100岁)
 * [5]: 步速归一化 (跌倒风险指标)
 * [6]: 握力归一化 (肌少症指标)
 * [7]: 6分钟步行距离归一化 (衰弱指标)
 * [8]: 性别编码 (0=女, 1=男)
 * [9]: 共病史编码 (0-1)
 */
export class HealthFeatureExtractor {
  static extractFeatures(healthData: HealthData): number[] {
    const bmi = healthData.weight / Math.pow(healthData.height / 100, 2);
    return [
      this.normalize(healthData.bloodPressureSystolic || 120, 60, 180),
      this.normalize(healthData.bloodPressureDiastolic || 80, 40, 120),
      this.normalize(healthData.bloodSugarFasting || 5.5, 3.9, 15.0),
      this.normalize(bmi, 15, 40),
      this.normalize(healthData.age || 50, 18, 100),
      this.normalize(healthData.gaitSpeed || 1.0, 0.3, 2.0),
      this.normalize(healthData.gripStrength || 30, 5, 60),
      this.normalize(healthData.walkDistance || 450, 100, 800),
      healthData.gender === 'male' ? 1 : 0,
      Math.min((healthData.chronicDiseases?.length || 0) / 3, 1)
    ];
  }
  
  private static normalize(value: number, min: number, max: number): number {
    return Math.max(0, Math.min(1, (value - min) / (max - min)));
  }
}
```

### 3.4 核心类设计

#### ImageAnalysisService.ets（影像分析核心服务）
```typescript
/**
 * 医学影像AI病灶识别与分析服务
 * 
 * 核心职责：
 * 1. 接收用户上传的医学影像
 * 2. 自动识别影像类型（X光/CT/皮肤照片等）
 * 3. 调度对应的AI模型进行病灶检测
 * 4. 生成病灶标注和分析报告
 * 5. 提供历史记录管理和查询
 */
export class ImageAnalysisService {
  private static instance: ImageAnalysisService;
  private modelManager: ModelManager;
  private preprocessor: ImagePreprocessor;
  private detector: LesionDetector;

  private constructor() {
    this.modelManager = new ModelManager();
    this.preprocessor = new ImagePreprocessor();
    this.detector = new LesionDetector();
  }

  public static getInstance(): ImageAnalysisService {
    if (!ImageAnalysisService.instance) {
      ImageAnalysisService.instance = new ImageAnalysisService();
    }
    return ImageAnalysisService.instance;
  }

  /**
   * 初始化服务（应用启动时调用一次）
   * 预加载常用模型到内存
   */
  async initialize(): Promise<void> {
    console.log('[ImageAnalysis] 初始化影像分析服务...');
    
    // 预加载最常用的模型（肺部和骨骼）
    await this.modelManager.preloadModels([
      ImagingType.CHEST_XRAY,
      ImagingType.BONE_XRAY
    ]);
    
    console.log('[ImageAnalysis] 服务初始化完成');
  }

  /**
   * 执行完整的影像分析流程
   * 
   * @param imageUri 本地图片URI或PixelMap
   * @param options 分析选项（影像类型、是否需要详细报告等）
   * @returns 完整的分析结果
   */
  async analyzeImage(
    imageUri: string,
    options?: AnalysisOptions
  ): Promise<ImageAnalysisResult> {
    const startTime = Date.now();
    
    try {
      console.log('[ImageAnalysis] 开始分析图片:', imageUri);

      // Step 1: 加载并解码图像
      const rawImage = await this.loadImage(imageUri);
      
      // Step 2: 自动识别影像类型（如果用户未指定）
      const imagingType = options?.imagingType || await this.detectImagingType(rawImage);
      console.log('[ImageAnalysis] 识别影像类型:', imagingType);

      // Step 3: 图像预处理（归一化、resize等）
      const preprocessedImage = await this.preprocessor.process(rawImage, imagingType);

      // Step 4: 加载对应AI模型（如未预加载）
      const model = await this.modelManager.getModel(imagingType);

      // Step 5: 执行病灶检测推理
      const detectionResult = await this.detector.detect(preprocessedImage, model, imagingType);

      // Step 6: 后处理（NMS非极大值抑制、置信度过滤）
      const filteredLesions = this.postProcess(detectionResult.lesions, options?.confidenceThreshold || 0.5);

      // Step 7: 生成分析报告
      const report = await this.generateAnalysisReport(filteredLesions, imagingType, rawImage);

      // 构建最终结果
      const result: ImageAnalysisResult = {
        analysisId: generateUUID(),
        imageUri: imageUri,
        imagingType: imagingType,
        lesions: filteredLesions,
        overallAssessment: this.calculateOverallAssessment(filteredLesions),
        report: report,
        processingTimeMs: Date.now() - startTime,
        modelVersion: model.version,
        confidenceThreshold: options?.confidenceThreshold || 0.5,
        analyzedAt: new Date().toISOString()
      };

      // Step 8: 保存分析记录到本地数据库
      await this.saveAnalysisRecord(result);

      console.log('[ImageAnalysis] 分析完成, 耗时:', result.processingTimeMs + 'ms');
      console.log('[ImageAnalysis] 发现病灶数量:', filteredLesions.length);

      return result;

    } catch (error) {
      console.error('[ImageAnalysis] 分析失败:', error.message);
      throw new Error('影像分析失败: ' + error.message);
    }
  }

  /**
   * 批量分析多张图片（用于对比分析）
   */
  async analyzeBatch(
    imageUris: string[],
    options?: AnalysisOptions
  ): Promise<ImageAnalysisResult[]> {
    const results: ImageAnalysisResult[] = [];
    
    for (let i = 0; i < imageUris.length; i++) {
      const result = await this.analyzeImage(imageUris[i], options);
      results.push(result);
    }
    
    return results;
  }

  /**
   * 获取历史分析记录
   */
  async getAnalysisHistory(limit: number = 20, offset: number = 0): Promise<AnalysisRecord[]> {
    const records = await RdbHelper.query(
      'SELECT * FROM ' + TABLE_IMAGING_ANALYSIS + 
      ' ORDER BY analyzed_at DESC LIMIT ? OFFSET ?',
      [String(limit), String(offset)]
    );

    const results: AnalysisRecord[] = [];
    while (records.goToNextRow()) {
      results.push({
        id: String(records.getLong(records.getColumnIndex('id'))),
        imageUri: records.getString(records.getColumnIndex('image_uri')),
        imagingType: records.getString(records.getColumnIndex('imaging_type')),
        lesionCount: records.getInt(records.getColumnIndex('lesion_count')),
        overallAssessment: records.getString(records.getColumnIndex('overall_assessment')) as AssessmentLevel,

    return results;

  // ========== 私有方法 ==========

  private async loadImage(imageUri: string): Promise<image.PixelMap> {
    const imageSource = image.createImageSource(imageUri);
    const pixelMap = await imageSource.createPixelMap();
    return pixelMap;
  }

  private async detectImagingType(pixelMap: image.PixelMap): Promise<ImagingType> {
    // 基于图像尺寸、宽高比、像素分布等特征初步判断类型
    // 这是一个简化版实现，实际可使用轻量级分类模型
    
    const imageSize = await this.getImageSize(pixelMap);
    const aspectRatio = imageSize.width / imageSize.height;

    // X光片通常是正方形或接近正方形 (0.9 - 1.1)
    if (aspectRatio >= 0.9 && aspectRatio <= 1.1 && imageSize.width >= 512) {
      return ImagingType.CHEST_XRAY;  // 默认假设为胸片
    }
    
    // 皮肤照片通常是竖版 (0.6 - 0.9)
    if (aspectRatio >= 0.6 && aspectRatio <= 0.9) {
      return ImagingType.SKIN_PHOTO;
    }

    // 默认返回胸部X光
    return ImagingType.CHEST_XRAY;
  }

  private postProcess(
    lesions: DetectedLesion[],
    threshold: number
  ): DetectedLesion[] {
    // 1. 过滤低置信度检测结果
    let filtered = lesions.filter(l => l.confidence >= threshold);

    // 2. NMS非极大值抑制（去除重叠的重复检测框）
    filtered = this.applyNMS(filtered, 0.45);

    // 3. 按置信度降序排序
    filtered.sort((a, b) => b.confidence - a.confidence);

    return filtered;
  }

  /**
   * NMS非极大值抑制算法
   */
  private applyNMS(boxes: DetectedLesion[], iouThreshold: number): DetectedLesion[] {
    if (boxes.length === 0) return [];

    const keep: DetectedLesion[] = [];
    const suppressed: boolean[] = new Array(boxes.length).fill(false);

    // 按置信度排序
    const sortedBoxes = [...boxes].sort((a, b) => b.confidence - a.confidence);

    for (let i = 0; i < sortedBoxes.length; i++) {
      if (suppressed[i]) continue;

      keep.push(sortedBoxes[i]);

      for (let j = i + 1; j < sortedBoxes.length; j++) {
        if (suppressed[j]) continue;

        const iou = this.calculateIoU(sortedBoxes[i].boundingBox, sortedBoxes[j].boundingBox);
        if (iou > iouThreshold) {
          suppressed[j] = true;
        }
      }
    }

    return keep;
  }

  private calculateIoU(box1: BoundingBox, box2: BoundingBox): number {
    const x1 = Math.max(box1.x, box2.x);
    const y1 = Math.max(box1.y, box2.y);
    const x2 = Math.min(box1.x + box1.width, box2.x + box2.width);
    const y2 = Math.min(box1.y + box1.height, box2.y + box2.height);

    const intersection = Math.max(0, x2 - x1) * Math.max(0, y2 - y1);
    const area1 = box1.width * box1.height;
    const area2 = box2.width * box2.height;
    const union = area1 + area2 - intersection;

    return union > 0 ? intersection / union : 0;
  }

  private calculateOverallAssessment(lesions: DetectedLesion[]): AssessmentLevel {
    if (lesions.length === 0) return AssessmentLevel.NORMAL;
    
    const maxConfidence = Math.max(...lesions.map(l => l.confidence));
    const highRiskCount = lesions.filter(l => l.riskLevel === RiskLevel.HIGH || l.riskLevel === RiskLevel.CRITICAL).length;

    if (highRiskCount >= 2 || maxConfidence >= 0.9) {
      return AssessmentLevel.REQUIRES_ATTENTION;  // 需要关注
    } else if (highRiskCount >= 1 || maxConfidence >= 0.7) {
      return AssessmentLevel.SUSPICIOUS;  // 可疑
    } else {
      return AssessmentLevel.PROBABLY_BENIGN;  // 可能良性
    }
  }

  private async generateAnalysisReport(
    lesions: DetectedLesion[],
    imagingType: ImagingType,
    originalImage: image.PixelMap
  ): Promise<AnalysisReport> {
    const generator = new AnalysisReportGenerator();
    return await generator.generate(lesions, imagingType, originalImage);
  }

  private async saveAnalysisRecord(result: ImageAnalysisResult): Promise<void> {
    await RdbHelper.insert(TABLE_IMAGING_ANALYSIS, {
      analysis_id: result.analysisId,
      image_uri: result.imageUri,
      imaging_type: result.imagingType,
      lesion_count: result.lesions.length,
      overall_assessment: result.overallAssessment,
      processing_time_ms: result.processingTimeMs,
      model_version: result.modelVersion,
      analyzed_at: result.analyzedAt,
      user_id: AppStorage.get<string>('userId') || ''
    });
  }

  private async getImageSize(pixelMap: image.PixelMap): Promise<{ width: number, height: number }> {
    const info = await pixelMap.getImageInfo();
    return { width: info.size.width, height: info.size.height };
  }
```

```typescript
/**
 * AI病灶检测引擎
 * 负责调用MindSpore Lite模型执行推理
 */
export class LesionDetector {

  /**
   * 执行病灶检测
   * 
   * @param preprocessedImage 预处理后的图像张量
   * @param model 已加载的MindSpore模型
   * @param imagingType 影像类型
   * @returns 原始检测结果（包含所有候选框）
   */
  async detect(
    preprocessedImage: Float32Array,
    model: mindspore.Model,
    imagingType: ImagingType
  ): Promise<{ lesions: RawDetectedLesion[], metadata: DetectionMetadata }> {
    
    const startTime = Date.now();

    try {
      // 构建输入Tensor（根据不同模型调整shape）
      const inputShape = this.getInputShape(imagingType);
      const inputTensor: mindspore.MSTensor = {
        dataType: mindspore.DataType.NUMBER_TYPE_FLOAT32,
        shape: inputShape,
        data: preprocessedImage.buffer
      };

      // 执行前向推理
      const outputTensors = await model.predict([inputTensor]);

      // 解析输出（根据模型类型不同，输出格式也不同）
      const rawResults = this.parseModelOutput(outputTensors, imagingType);

      // 将原始输出转换为统一的病灶格式
      const lesions: RawDetectedLesion[] = rawResults.map((raw, index) => ({
        id: 'lesion_' + index,
        classId: raw.classId,
        className: this.getClassName(imagingType, raw.classId),
        confidence: raw.confidence,
        bbox: {
          x: raw.bbox[0],
          y: raw.bbox[1],
          width: raw.bbox[2] - raw.bbox[0],
          height: raw.bbox[3] - raw.bbox[1]
        },
        additionalScores: raw.scores || {}
      }));

      const metadata: DetectionMetadata = {
        inferenceTimeMs: Date.now() - startTime,
        totalCandidates: lesions.length,
        modelInputSize: inputShape,
        imagingType: imagingType
      };

      return { lesions, metadata };

    } catch (error) {
      console.error('[LesionDetector] 推理失败:', error.message);
      throw error;
    }
  }

  /**
   * 获取模型的输入shape
   */
  private getInputShape(imagingType: ImagingType): number[] {
    switch (imagingType) {
      case ImagingType.CHEST_XRAY:
      case ImagingType.BONE_XRAY:
        return [1, 3, 512, 512];  // ResNet/EfficientNet标准输入
      
      case ImagingType.SKIN_PHOTO:
        return [1, 3, 224, 224];  // MobileNetV3标准输入
      
      case ImagingType.EYE_FUNDUS:
        return [1, 3, 512, 512];  // DenseNet输入
      
      case ImagingType.BREAST_ULTRASOUND:
        return [1, 3, 224, 224];  // VGG16输入
      
      default:
        return [1, 3, 224, 224];
    }
  }

  /**
   * 解析模型输出（不同架构输出格式不同）
   * YOLO/SSD输出: [batch, num_detections, 6] (x1,y1,x2,y2,conf,class)
   * Faster R-CNN输出: boxes + scores + classes 分离的张量
   */
  private parseModelOutput(
    outputs: mindspore.MSTensor[],
    imagingType: ImagingType
  ): RawDetectionResult[] {
    
    const results: RawDetectionResult[] = [];

    // 根据影像类型选择解析策略
    if (this.isYOLOStyleOutput(imagingType)) {
      // YOLO风格输出：单个张量 [N, 6]
      const output = outputs[0];
      const data = new Float32Array(output.data);
      const numDetections = output.shape[0];

      for (let i = 0; i < numDetections; i++) {
        const offset = i * 6;
        const confidence = data[offset + 4];
        
        if (confidence > 0.01) {  // 先不过滤，留给后处理
          results.push({
            classId: Math.round(data[offset + 5]),
            confidence: confidence,
            bbox: [
              data[offset + 0],
              data[offset + 1],
              data[offset + 2],
              data[offset + 3]
            ]
          });
        }
      }
    } else {
      // Faster R-CNN风格：boxes + scores + classes
      const boxes = new Float32Array(outputs[0].data);
      const scores = new Float32Array(outputs[1].data);
      const classes = new Float32Array(outputs[2].data);
      const numDetections = outputs[0].shape[0];

      for (let i = 0; i < numDetections; i++) {
        if (scores[i] > 0.01) {
          results.push({
            classId: Math.round(classes[i]),
            confidence: scores[i],
            bbox: [
              boxes[i * 4 + 0],
              boxes[i * 4 + 1],
              boxes[i * 4 + 2],
              boxes[i * 4 + 3]
            ]
          });
        }
      }
    }

    return results;
  }

  private isYOLOStyleOutput(imagingType: ImagingType): boolean {
    // 胶片和皮肤检测使用YOLO架构
    return [
      ImagingType.CHEST_XRAY,
      ImagingType.SKIN_PHOTO
    ].includes(imagingType);
  }

  /**
   * 将classId转换为人类可读的病灶类型名称
   */
  private getClassName(imagingType: ImagingType, classId: number): string {
    const classNames: Record<string, string[]> = {
      [ImagingType.CHEST_XRAY]: ['正常', '肺结节', '肺炎', '气胸', '心脏肥大', '胸腔积液'],
      [ImagingType.BONE_XRAY]: ['正常', '骨折', '骨质疏松', '关节病变', '骨肿瘤', '脱位'],
      [ImagingType.SKIN_PHOTO]: ['正常', '黑色素瘤', '良性痣', '湿疹', '银屑病', '基底细胞癌'],
      [ImagingType.EYE_FUNDUS]: ['正常', '糖网病变', '青光眼', '黄斑变性', '白内障', '视网膜脱离'],
      [ImagingType.BREAST_ULTRASOUND]: ['正常', '良性肿块', '可疑肿块', '恶性肿块', '囊肿', '钙化']
    };

    const names = classNames[imagingType] || ['未知'];
    return names[classId] || ('类别' + classId);
  }
}
```
