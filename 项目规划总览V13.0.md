```typescript
// Login.ets - 登录页面核心逻辑
@Entry
@Component
struct Login {
  @State username: string = '';
  @State password: string = '';

  async handleLogin() {
    const response = await HttpUtil.post('/api/user/login', {
      username: this.username,
      password: this.password
    });
    if (response.code === 200) {
      AppStorage.set('token', response.data.token);
      router.replaceUrl({ url: 'pages/HomePage' });
    }
  }
}
```

```java
// UserController.java - 后端登录API
@PostMapping("/login")
public Result<?> login(@RequestBody LoginDTO loginDTO) {
    User user = userService.getByUsername(loginDTO.getUsername());
    if (!BCryptUtil.matches(loginDTO.getPassword(), user.getPassword())) {
        return Result.error("密码错误");
    }
    String token = JwtUtil.generateToken(user.getId(), user.getRole());
    return Result.success(Map.of("token", token, "user", user));
}
```

**DosageCalc.ets 核心算法**:

```typescript
interface DosageResult {
  adultDosage: string;
  childDosage: string;
  unit: string;
  formula: string;
}

calculateDosage(): void {
  let baseDosage: number = 10;

  switch (this.dosageType) {
    case '汤剂':
      baseDosage = this.weight * 0.15; // 汤剂标准：体重×0.15g/kg
      break;
    case '丸剂':
      baseDosage = this.weight * 0.08 * 0.6; // 丸剂吸收率较低
      break;
    case '散剂':
      baseDosage = this.weight * 0.1; // 散剂起效快但需谨慎
      break;
  }

  const childMultiplier = this.age / (this.age + 12); // Young公式
  const childDosage = baseDosage * childMultiplier;

  if (this.isPregnant && this.dosageType === '汤剂') {
    baseDosage *= 0.7; // 孕妇减量30%
  }

  this.result = {
    adultDosage: `${baseDosage.toFixed(1)}`,
    childDosage: childDosage > 0 ? `${childDosage.toFixed(1)}` : '不适用',
    unit: 'g',
    formula: `${this.dosageType}标准${this.isPregnant ? '(孕妇减量)' : ''}`
  };
}
```
