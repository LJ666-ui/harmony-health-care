  private reporter: ReportGenerator

  async processImage(imageSource: image.ImageSource): Promise<ProcessResult> {
    const result = new ProcessResult()

    // Step1: 预处理
    result.step1 = await this.preprocessor.process(imageSource)
    this.emitProgress('step1', 100)

    // Step2: 色彩恢复
    result.step2 = await this.colorRestorer.restore(result.step1.output)
    this.emitProgress('step2', 100)

    // Step3: 超分辨率
    result.step3 = await this.superRes.enhance(result.step2.output, 4)
    this.emitProgress('step3', 100)

    // Step4: OCR识别(可与Step3并行)
    result.ocrResult = await this.ocr.recognize(result.step3.output)

    return result
  }
}
